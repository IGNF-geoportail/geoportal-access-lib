Utilisation du Service de Calcul Altimétrique

Avec les paramètres WPS "Profil" en mode POST

          
              var options = {
                apiKey : "{{ apikey }}",
                httpMethod : "POST",
                protocol : "XHR",
                onSuccess : function (response) {
                    print(response.elevations);
                },
                // spécifique au service
                positions : [{lon:1.25, lat:47.48}, {lon:1.35, lat:47.50}],
                outputFormat : "xml", // json|xml
                sampling : 3,
                api : "REST", // REST|WPS
                zonly : false // false|true
            };
            Gp.Services.getAltitude(options);
